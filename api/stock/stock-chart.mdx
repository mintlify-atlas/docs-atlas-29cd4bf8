## Overview

StockChart is the main class for creating interactive financial charts in amCharts 5. It provides a complete solution for displaying stock data with multiple panels, technical indicators, comparison series, and drawing tools.

## Class Reference

### Creating a StockChart

```typescript
import { StockChart } from "@amcharts/amcharts5/stock";

const stockChart = root.container.children.push(
  StockChart.new(root, {
    stockSeries: mainSeries,
    volumeSeries: volumeSeries
  })
);
```

## Settings

<ParamField path="stockSeries" type="XYSeries">
  Main value series used for calculations, indicators, and annotations.
  
  [Learn more about setting main series](https://www.amcharts.com/docs/v5/charts/stock/#Setting_main_series)
</ParamField>

<ParamField path="volumeSeries" type="XYSeries">
  Main volume series for the chart.
  
  [Learn more about setting main series](https://www.amcharts.com/docs/v5/charts/stock/#Setting_main_series)
</ParamField>

<ParamField path="autoSetPercentScale" type="boolean" default="true">
  When `true`, the chart automatically switches to percent scale when comparison series are added, and exits when all comparisons are removed.
  
  [Learn more about percent mode](https://www.amcharts.com/docs/v5/charts/stock/percent-mode/)
</ParamField>

<ParamField path="percentScaleSeriesSettings" type="Partial<IXYSeriesSettings>">
  Settings applied to the main value series when the chart is in percent scale mode.
  
  [Learn more about configuring percent mode](https://www.amcharts.com/docs/v5/charts/stock/percent-mode/#Configuring)
</ParamField>

<ParamField path="percentScaleValueAxisSettings" type="Partial<IValueAxisSettings>">
  Settings applied to the ValueAxis of the main series when the chart is in percent scale mode.
  
  [Learn more about configuring percent mode](https://www.amcharts.com/docs/v5/charts/stock/percent-mode/#Configuring)
</ParamField>

<ParamField path="stockPositiveColor" type="Color">
  Color for columns/candles on the main value series where the open value is lower than or equal to the close value.
  
  [Learn more about positive/negative colors](https://www.amcharts.com/docs/v5/charts/stock/#Positive_negative_colors)
</ParamField>

<ParamField path="stockNegativeColor" type="Color">
  Color for columns/candles on the main value series where the open value is higher than the close value.
  
  [Learn more about positive/negative colors](https://www.amcharts.com/docs/v5/charts/stock/#Positive_negative_colors)
</ParamField>

<ParamField path="volumePositiveColor" type="Color">
  Color for volume columns where the close value is higher than or equal to the open value.
  
  [Learn more about positive/negative colors](https://www.amcharts.com/docs/v5/charts/stock/#Positive_negative_colors)
</ParamField>

<ParamField path="volumeNegativeColor" type="Color">
  Color for volume columns where the close value is lower than the open value.
  
  [Learn more about positive/negative colors](https://www.amcharts.com/docs/v5/charts/stock/#Positive_negative_colors)
</ParamField>

<ParamField path="drawingSelectionEnabled" type="boolean" default="false" since="5.9.1">
  When `true`, all drawings are selectable and can be moved, rotated, or deleted.
</ParamField>

<ParamField path="erasingEnabled" type="boolean" default="false" since="5.9.2">
  When `true`, clicking on drawings deletes them.
</ParamField>

<ParamField path="hideDrawingGrips" type="boolean" default="false" since="5.10.6">
  When `true`, drawing grips are hidden when the StockChart is not in drawing mode.
</ParamField>

<ParamField path="autoHidePanelControls" type="boolean" default="false" since="5.13.0">
  When `true`, panel controls automatically hide when the mouse is not over the chart.
</ParamField>

## Methods

### setPercentScale()

Enables or disables percent scale mode.

```typescript
setPercentScale(percentScale?: boolean): void
```

<ParamField path="percentScale" type="boolean">
  Whether to enable percent scale mode. If not set, automatically determined based on compared series.
</ParamField>

In percent scale mode, `percentScaleSeriesSettings` and `percentScaleValueAxisSettings` are applied to the main series and its Y axis.

### addComparingSeries()

Adds a comparison series to the chart.

```typescript
addComparingSeries(series: XYSeries): XYSeries
```

<ParamField path="series" type="XYSeries" required>
  The series to add for comparison.
</ParamField>

<ResponseField name="return" type="XYSeries">
  The same series that was added.
</ResponseField>

### removeComparingSeries()

Removes a comparison series from the chart.

```typescript
removeComparingSeries(series: XYSeries): void
```

<ParamField path="series" type="XYSeries" required>
  The series to remove.
</ParamField>

### getVolumeColor()

Returns a color for volume based on current and previous close values.

```typescript
getVolumeColor(
  dataItem: DataItem<IXYSeriesDataItem>,
  negativeColor?: Color,
  positiveColor?: Color
): Color | undefined
```

<ParamField path="dataItem" type="DataItem<IXYSeriesDataItem>" required>
  Target data item.
</ParamField>

<ParamField path="negativeColor" type="Color">
  Color to use when close is lower than previous close (defaults to `volumeNegativeColor`).
</ParamField>

<ParamField path="positiveColor" type="Color">
  Color to use when close is greater than or equal to previous close (defaults to `volumePositiveColor`).
</ParamField>

<ResponseField name="return" type="Color | undefined">
  The calculated volume color.
</ResponseField>

### getControl()

Returns the first StockControl of a specific type.

```typescript
getControl(type: string): StockControl | undefined
```

<ParamField path="type" type="string" required>
  Control name (e.g., "IndicatorControl", "DrawingControl").
</ParamField>

<ResponseField name="return" type="StockControl | undefined">
  The control instance, or undefined if not found.
</ResponseField>

@since 5.7.0

## Example

```typescript
import * as am5 from "@amcharts/amcharts5";
import * as am5stock from "@amcharts/amcharts5/stock";
import * as am5xy from "@amcharts/amcharts5/xy";

const root = am5.Root.new("chartdiv");

// Create main stock panel
const mainPanel = root.container.children.push(
  am5stock.StockPanel.new(root, {})
);

// Create axes
const valueAxis = mainPanel.yAxes.push(
  am5xy.ValueAxis.new(root, {
    renderer: am5xy.AxisRendererY.new(root, {})
  })
);

const dateAxis = mainPanel.xAxes.push(
  am5xy.GaplessDateAxis.new(root, {
    baseInterval: { timeUnit: "day", count: 1 },
    renderer: am5xy.AxisRendererX.new(root, {})
  })
);

// Create series
const valueSeries = mainPanel.series.push(
  am5xy.CandlestickSeries.new(root, {
    xAxis: dateAxis,
    yAxis: valueAxis,
    valueXField: "Date",
    valueYField: "Close",
    openValueYField: "Open",
    highValueYField: "High",
    lowValueYField: "Low"
  })
);

// Create stock chart
const stockChart = root.container.children.push(
  am5stock.StockChart.new(root, {
    stockSeries: valueSeries,
    stockPositiveColor: am5.color(0x00ff00),
    stockNegativeColor: am5.color(0xff0000)
  })
);

stockChart.panels.push(mainPanel);

// Add indicator
const ma = stockChart.indicators.push(
  am5stock.MovingAverage.new(root, {
    stockChart: stockChart,
    stockSeries: valueSeries,
    period: 50,
    seriesColor: am5.color(0x0000ff)
  })
);
```
